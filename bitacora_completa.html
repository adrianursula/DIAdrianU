v0.2 BETA - Visualización Funcional

12 diciembre 2025

🐛 Problema Resuelto

El usuario reportó: "no puedo subir o arrastrar y soltar un archivo para visualizarlo"

Causa: v0.1 solo detectaba archivos pero no los leía ni visualizaba.

✨ Solución: FileReader API

// Para imágenes: Base64
const reader = new FileReader();
reader.readAsDataURL(file); // Convierte a Data URL

// Para texto/código: String
reader.readAsText(file); // Lee como texto plano

💻 Implementación: Conversión de Imágenes

// Crear canvas
const img = new Image();
img.src = processedContent; // Data URL
await new Promise(resolve => img.onload = resolve);

const canvas = document.createElement('canvas');
canvas.width = img.width;
canvas.height = img.height;

// Dibujar imagen
const ctx = canvas.getContext('2d');
ctx.drawImage(img, 0, 0);

// Exportar a nuevo formato
canvas.toBlob(blob => {
    downloadFile(blob, 'converted.png', 'image/png');
}, 'image/png', 0.95); // Calidad 95%

💻 Implementación: JSON a CSV

const json = JSON.parse(processedContent);
let data = Array.isArray(json) ? json : [json];

// Headers
const headers = Object.keys(data[0]);
let csv = headers.join(',') + '\n';

// Rows con escapado de comas
data.forEach(row => {
    const values = headers.map(header => {
        const val = row[header];
        return typeof val === 'string' && val.includes(',') 
            ? `"${val}"` 
            : val;
    });
    csv += values.join(',') + '\n';
});

⚡ Función Universal de Descarga

function downloadFile(blob, filename, mimeType) {
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = filename;
    document.body.appendChild(a);
    a.click();
    document.body.removeChild(a);
    URL.revokeObjectURL(url);
    alert(`✅ ${filename} descargado`);
}
